struct ponto {
       double x; // endereço na coordenada X
       double y; // endereço na coordenada Y
};

struct lista {
       int tam; // variável pra controlar o tamanho da lista
       struct ponto v[100]; // Área de armazenamento da lista
};

void iniciaLista (lista *l){
     l->tam = 0;    
}

void inserir(lista *l, ponto p){
     if (l->tam == 100){
         cout << "\n\nERRO! lista cheia...\n";         
     }
     l->v[l->tam] = p;
     (l->tam)++;
     return;
}

double dist (ponto a, ponto b){
       double q1, q2, resp;
       q1 = a.x - b.x;
       q2 = a.y - b.y;
       q1 = q1 * q1;
       q2 = q2 * q2;
       resp = sqrt (q1 + q2);
       return resp;
}

double caminho (lista l){
	int i; // para controlar o laço
	double r=0; // para acumular a resposta
	for (i = 0; i < ((l.tam)-1); i++){
//		cout << "\nDistancia entre ponto " << (i+1) << " e ponto " << (i+2) << ": ";
//		cout << rt;
		r = r + dist(l.v[i],l.v[i+1]);
    }
    return r;   
}
