#include <iostream>
#include <cstdlib>
using namespace std;

void inserir(int v[], int &n, int e){
	if (40 == n){
		cout << "ERRO: Lista cheia.";
	}else{
		v[n] = e;
		n++;
	}
}

void percorrer(int v[], int n){
	for (int i = 0; i < n; i++)
		cout <<  v[i] << " ";
}

bool troca (int v[], int a, int b){
     int aux;
     aux = v[a];
     v[a] = v[b];
     v[b] = aux;
     return true;
}

void ordenarBolha(int v[], int n){
	int i,fim=n-1,aux;
	bool trocou = true;
	while (trocou){
		trocou = false;
		for (i = 0; i < fim; i++){
			if (v[i] > v[i+1]){
				trocou = troca(v,i,(i+1));
			}
		}
		fim--;
	} // fim while
}// fim da função


void ordenaSelecao(int v[], int n){
	int i, j, menor, aux;
	for (j = 0; j < n-1; j++){
		menor = j;
		for (i = j+1; i < n; i++){
			if (v[i] < v[menor]) menor = i;
		}
		troca(v,j,menor);
	}
}


void ordenaInsercao(int v[], int n){
	int i, j, aux;
	for (j = 1; j < n; j++)
		for (i=j; i > 0 && v[i-1]> v[i]; i--){
            troca(v,i,(i-1));
		}
}

// Prog. Principal
int main(){
	int v[40], tam, i, temp;
        tam = 0;
	for (i = 0; i < 15; i++){
		cout << "Digite um numero: ";
		cin >> temp;
		inserir(v, tam, temp);
	}
	cout << "\n\n Imprimindo antes de ordenar: ";
	percorrer(v,tam);
	ordenaInsercao(v,tam);
	cout << "\n\n Imprimindo depois de ordenar: ";
	percorrer(v,tam);
	cout << "\n\n.";
	system("pause");
	return 0;
}
